from PIL import Image def create_sticker(input_path, output_path): # Открываем изображение img = Image.open(input_path) # Создаем новое изображение для стикера sticker = Image.new('RGBA', img.size, (255, 255, 255, 0)) # Вставляем оригинальное изображение sticker.paste(img, (0, 0)) # Сохраняем результат sticker.save(output_path, "PNG") # Пример использования input_image_path = "input_image.jpg" output_sticker_path = "output_sticker.png" create_sticker(input_image_path, output_sticker_path)
